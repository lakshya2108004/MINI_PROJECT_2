\begin{frame}{Problem Statement}
\begin{block}{Current Issues}
\begin{itemize}
\item Study materials are scattered across different platforms.
\item Students struggle to find reliable notes and PDFs.
\item No centralized system for sharing academic resources.
\end{itemize}
\end{block}

\begin{alertblock}{Goal}
Create a centralized web platform where students can upload, search, and download study materials easily.
\end{alertblock}

\end{frame}

%------------------------------------------------

\begin{frame}{Proposed Solution}

\begin{columns}

\column{0.5\textwidth}

\textbf{Core Idea}

\begin{itemize}
\item Upload study materials
\item Download PDFs
\item Organized by subject and semester
\item Secure authentication
\end{itemize}

\column{0.5\textwidth}

\begin{block}{Benefits}
\begin{itemize}
\item Easy access to resources
\item Collaboration among students
\item Organized academic content
\end{itemize}
\end{block}

\end{columns}

\end{frame}

\begin{frame}{Authentication and Authorization}

\begin{block}{Authentication Methods}
\begin{itemize}
\item Users can sign up using email and password.
\item Passwords are securely hashed before storing in the database.
\item Users can also log in using OAuth providers (e.g., Google login).
\end{itemize}
\end{block}

\begin{block}{JWT Based Authorization}
\begin{itemize}
\item After successful login, the server generates a JSON Web Token (JWT).
\item The token is sent to the client and stored securely.
\item Every protected API request includes the JWT for verification.
\end{itemize}
\end{block}

\begin{exampleblock}{Benefits}
\begin{itemize}
\item Secure authentication
\item Stateless session management
\item Easy integration with third-party login systems
\end{itemize}
\end{exampleblock}

\end{frame}
